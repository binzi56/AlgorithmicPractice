//解法一：
//每次将前n-1个数向后移动一位，并将原来最后一位与第一位交换
void rotate(vector<int>& nums, int k) {
    if(nums.empty()) return;

    int temp = 0, n = nums.size();
    for(int i = 0; i < k % n; i++){
        temp = nums[n - 1];
        for(int j = n - 1; j > 0; j--){
            nums[j] = nums[j - 1];
        }
        nums[0] = temp;
    }
}

//解法二：
// 三重反转
// 把一个数组的右边一部分移动到左边相当于：
// 把左部分翻转
// 把右部分翻转
// 最后把整体翻转
void rotate(vector<int>& nums, int k) {
    if(nums.empty()) return;

    int n = k % nums.size();
    reverse(nums.begin(), nums.end() - n);
    reverse(nums.end() - n, nums.end());
    reverse(nums.begin(), nums.end());
}
